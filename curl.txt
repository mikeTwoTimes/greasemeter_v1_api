# Places
  # Map markers request
    curl -X GET "https://api.greasemeter.live/v1/places/map?lat=float64&lng=float64&latDelta=float64&lngDelta=float64" \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Map markers response (200)
    [
      {
        "id": int,
        "point": geoJSONPoint
      }
    ]

  # List view request
    curl -X GET "https://api.greasemeter.live/v1/places/list?lat=float64&lng=float64&latDelta=float64&lngDelta=float64&page=int&limit=int" \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # List view response (200)
    {
      "data": [
        "id": int
    	"name": string
        "address": string
        "rating": int
      ],
      "more": bool
    }  

  # Search request
    curl -X GET "https://api.greasemeter.live/v1/places/search?term=string&lat=float64&lng=float64" \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Search response (200)
    [
      {
        "id": int,
        "name": string,
        "address": string
      }
    ]

  # Meta details for search results and bookmarks
    curl -X GET https://api.greasemeter.live/v1/places/{int}/meta \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Detail response (200)
    {
      "point": geoJSONPoint
      "images": [string]
      "rating": float32
    }

  # Details for map markers
    curl -X GET https://api.greasemeter.live/v1/places/{int}/map \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Detail response (200)
    {
      "address": string
      "name": string
      "images": [string]
      "rating": float32
    }

  # Details for a place listing
    curl -X GET https://api.greasemeter.live/v1/places/{int}/list \
    -H "Content-Type: application/json" \
    -w "\nHTTP Status: %{http_code}\n"

  # Detail response (200)
    {
      "point": geoJSONPoint
      "images": [string]
    }
